111 XMLObject::XMLObject(const Namespace& xmlns, const ClassName& className, const xercesc::DOMElement* element, const Attributes& defaultAttributes) : xmlns(xmlns), className(className) {
112 
113  prefix = xercesc::XMLString::transcode(element->getPrefix());
114  elementName = xercesc::XMLString::transcode(element->getLocalName());
115 
116  // set attributes
117  xercesc::DOMNamedNodeMap* elementAttributes = element->getAttributes();
118  for (XMLSize_t i = 0; i < elementAttributes->getLength(); i++) {
119  xercesc::DOMNode* item = elementAttributes->item(i);
120 
121  AttributeName attributeName = xercesc::XMLString::transcode(item->getLocalName());
122  // get namespace from atrribute or parent element
123  Namespace attributeXmlns = item->getNamespaceURI() ? xercesc::XMLString::transcode(item->getNamespaceURI()) : xmlns;
124  Namespace attributePrefix = item->getPrefix() ? xercesc::XMLString::transcode(item->getPrefix()) : "";
125  Value attributeValue((std::string)xercesc::XMLString::transcode(item->getNodeValue()));
126  attributes.push_back( { attributeXmlns, attributePrefix, attributeName, attributeValue } );
127  }
128 
129  // add defaults for missing attributes
130  for ( auto& defaultAttribute : defaultAttributes ) {
131  if ( !getOptionalAttributeByName(defaultAttribute.name) ) {
132  attributes.push_back(defaultAttribute);
133  }
134  }
135 
136  // set children
137  for (xercesc::DOMElement *childElement = element->getFirstElementChild(); childElement; childElement = childElement->getNextElementSibling()) {
138  children.push_back(std::unique_ptr<XMLObject>(createObject(childElement)));
139  }
140 
141  if ( children.empty() ) {
142  textContent = xercesc::XMLString::transcode(element->getTextContent());
143  }
144 }
