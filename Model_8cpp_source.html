489 void Model::createChildNodes(Scope* scope) {
490  // add flow nodes (except boundary events)
491  for (XML::bpmn::tFlowNode& flowNode : scope->element->getChildren<XML::bpmn::tFlowNode>() ) {
492  if ( auto subProcess = flowNode.is<XML::bpmn::tSubProcess>();
493  subProcess &&
494  subProcess->triggeredByEvent.has_value() &&
495  (bool)subProcess->triggeredByEvent->get().value
496  ) {
497  scope->add(createEventSubProcess(subProcess,scope));
498  }
499  else if ( !flowNode.is<XML::bpmn::tBoundaryEvent>() ) {
500  // create node according to element type
501  scope->add(createFlowNode(&flowNode,scope));
502  }
503  }
504  // add boundary events
505  for (XML::bpmn::tFlowNode& flowNode: scope->element->getChildren<XML::bpmn::tFlowNode>() ) {
506  if ( auto boundaryEvent = flowNode.is<XML::bpmn::tBoundaryEvent>(); boundaryEvent ) {
507  scope->add(createBoundaryEvent(boundaryEvent,scope));
508  }
509  }
510  // recurse
511  for ( auto& childNode: scope->childNodes ) {
512  if ( auto childScope = childNode->represents<Scope>(); childScope ) {
513  createChildNodes(childScope);

542  if ( flowNode->parent ) {

544  for ( auto& inflow : flowNode->element->incoming ) {
545  for (auto& sequenceFlow : flowNode->parent->sequenceFlows ) {

552

568  id.value().get().value.value == sequenceFlow->id
569  ) {

581  }

625 void Model::createCompensationReferences(Scope* scope) {
626  std::vector<CompensateThrowEvent*> compensateThrowEvents;
627  std::vector<Activity*> compensatedActivities;
628 
629  for ( auto& childNode: scope->childNodes ) {
630  if ( auto compensateThrowEvent = childNode->represents<CompensateThrowEvent>(); compensateThrowEvent ) {
631  compensateThrowEvents.push_back(compensateThrowEvent);
632  }
633  }
634 
635 
636  auto context = scope;
637  if ( auto eventSubProcess = scope->represents<EventSubProcess>();
638  eventSubProcess &&
639  scope->startEvents.size() == 1 &&
640  scope->startEvents.front()->represents<CompensateStartEvent>()
641  ) {
642  // compensation throw events in compensation event subprocess trigger
643  // compensation of activities in parent scope
644  context = eventSubProcess->parent;
645  }
646 
647  for ( auto& childNode: context->childNodes ) {
648  if ( auto compensateBoundaryEvent = childNode->represents<CompensateBoundaryEvent>(); compensateBoundaryEvent ) {
649  // add activity with compensate event attached to the boundary
650  auto activity = compensateBoundaryEvent->attachedTo->as<Activity>();
651  compensatedActivities.push_back(activity);
652  }
653  else if ( auto subprocess = childNode->represents<SubProcess>();
654  // add subprocess with compensation event subprocess
655  subprocess &&
656  subprocess->compensationEventSubProcess
657  ) {
658  compensatedActivities.push_back(subprocess);
659  }
660  }
661 
662 
663  // determine activity to be compensated for compensate throw event
664  for ( auto compensateThrowEvent : compensateThrowEvents ) {
665  auto& eventDefinition = compensateThrowEvent->element->getChildren<XML::bpmn::tCompensateEventDefinition>()[0].get();
666 
667  std::string activityRef;
668 
669  // try to get activityRef from event definition
670  if ( eventDefinition.activityRef.has_value() ) {
671  activityRef = eventDefinition.activityRef.value().get().value.value;
672  }
673 
674  if ( !activityRef.empty() ) {
675  auto it = std::find_if(
676  compensatedActivities.begin(),
677  compensatedActivities.end(),
678  [&activityRef](const Activity* activity) {
679  return activity && activity->id == activityRef;
680  });
681  if ( it == compensatedActivities.end() ) {
682  throw std::runtime_error("Model: Cannot find activity '" + activityRef + "' to be compensated");
683  }
684  compensateThrowEvent->activity = *it;
685  }
686  else if ( compensateThrowEvent->name.has_value() && !compensateThrowEvent->name.value().empty() ) {
687  // use node name as fallback
688  auto activityName = compensateThrowEvent->name.value();
689  auto it = std::find_if(
690  compensatedActivities.begin(),
691  compensatedActivities.end(),
692  [&activityName](const Activity* activity) {
693  return activity && activity->name.has_value() && activity->name.value() == activityName;
694  });
695  if ( it == compensatedActivities.end() ) {
696  throw std::runtime_error("Model: Cannot find activity with name '" + activityName + "' to be compensated");
697  }
698  compensateThrowEvent->activity = *it;
699  }
700 
701  }
